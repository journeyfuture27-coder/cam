function createMarker(point) { const container = document.createElement('div'); container.style.cssText = `display: flex; align-items: center; justify-content: center; width: 38px; height: 38px; transform: translate(-50%, -50%); cursor: pointer; filter: drop-shadow(0 2px 6px rgba(0,0,0,0.3)); transition: transform 0.2s;`; const icon = document.createElement('div'); icon.innerHTML = '📹'; icon.style.cssText = `font-size: 28px; width: 38px; height: 38px; display: flex; align-items: center; justify-content: center; background: white; border-radius: 50%; border: 2px solid #ff4444; box-shadow: 0 2px 8px rgba(0,0,0,0.2); transition: all 0.2s;`; container.appendChild(icon); // Данные для отображения const title = point.title; const description = point.description || "—"; const dateStr = point.entryDate || "—"; const sourceStr = point.source || "—"; const directionStr = point.direction !== null ? point.direction + "°" : "—"; const rangeStr = point.range !== null ? point.range + " м" : "—"; // Текст в нижней панели (добавляем Direction и Range) const fullInfo = `${title}\n${description}\n📅 ${dateStr} | 👤 ${sourceStr}\n🎯 Направление: ${directionStr} | Дальность: ${rangeStr}`; // Текст в alert const alertInfo = `${title}\n\nОписание: ${description}\n\nДата: ${dateStr}\nИсточник: ${sourceStr}\nНаправление: ${directionStr}\nДальность: ${rangeStr}`; container.addEventListener('mouseenter', () => { updateInfo(fullInfo); icon.style.border = '3px solid #ff8888'; icon.style.transform = 'scale(1.05)'; container.style.transform = 'translate(-50%, -50%) scale(1.05)'; }); container.addEventListener('mouseleave', () => { icon.style.border = '2px solid #ff4444'; icon.style.transform = 'scale(1)'; container.style.transform = 'translate(-50%, -50%) scale(1)'; }); container.addEventListener('click', (e) => { e.stopPropagation(); updateInfo(fullInfo); alert(alertInfo); selectedPoint = point; const zoom = currentMap.zoom; if (zoom >= 15 && point.direction && point.range && point.range > 0) { drawCone(currentMap, point.coordinates, point.direction, point.range); } else { removeCone(currentMap); } }); return new ymaps3.YMapMarker({ coordinates: point.coordinates }, container); }
